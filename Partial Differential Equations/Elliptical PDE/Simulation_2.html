def grid(N_i,N_j,value_interior,T1):
    # Array of zeros of correct dimensions
    array = np.zeros((N_j,N_i))
    # Boundary Value at y=0, all others boundary values set to 0 by default
    array[0,:] = T1
    # Assigning interior point values
    array[1:-1,1:-1] = value_interior
    return array

Stopping Criteria¶

In [5]:
@jit(nopython=True)
def converged(N_i,N_j,old,new,tol):
    residual_array = np.abs(new[1:-1,1:-1] - old[1:-1,1:-1])
    residual_sum = np.sum(residual_array)
    # tol = (N_i - 2)*(N_j - 2) * 1.3e-5
    
    if residual_sum < tol:
        return True,residual_sum
    else:
        return False,residual_sum

@jit(nopython=True)
def jacobi(N_i,N_j,beta,array,iter_max,tol):
    conv = False
    iter = 0
    residuals = []
    # Initial staring array i.e n_0 for 0th iteration
    u_k = array.copy()
    # First iteration inherits BC for previous step
    u_k_1 = u_k.copy()

    # Iterating till either convergence met or max number of iterations
    while not conv and iter <iter_max:
        iter += 1

        u_k_1[1:-1,1:-1] = 1/(2*(1+beta**2)) * (
            u_k[:-2,1:-1] + u_k[2:,1:-1] + (beta**2 * (u_k[1:-1,:-2] + u_k[1:-1,2:]))
        )

        # Checking convergence
        conv , residual = converged(N_i,N_j,u_k,u_k_1,tol)
        residuals.append(residual)

        u_k = u_k_1.copy()
        
    return u_k_1 , iter , residuals

def point_gauss(N_i,N_j,beta,array,iter_max,tol):

    u_k = array.copy()
    # BC for next step
    u_k_1 = array.copy()
    iter = 0
    residuals = []

            for j in range(1,N_i-1):

                u_k_1[i-1,j] + u_k[i+1,j] + beta**2 * (u_k_1[i,j-1]+u_k[i,j+1])

        conv , residual = converged(N_i,N_j,u_k,u_k_1,tol)

        u_k = u_k_1.copy()

def line_gauss_j(N_i,N_j,beta,array,iter_max,tol):
    conv = False
    u_k = array.copy()
    # BC for next step
    u_k_1 = array.copy()
    B = np.diag((-2*(1+beta**2)) * np.ones(N_i-2)) + np.diag(np.ones(N_i-3), -1) + np.diag(np.ones(N_i-3), 1)
    precondition = spilu(csc_matrix(B))
    iter = 0
    residuals=[]

    while not conv and iter < iter_max:
        iter += 1 
        for j in range(1,N_j-1):
            G = -beta**2 * (u_k_1[j-1,1:-1] + u_k[j+1,1:-1])
            G[0] -= u_k_1[j,0]
            G[-1] -= u_k_1[j,-1]
            # u_k_1[j,1:-1] = solve(B,G)
            u_k_1[j,1:-1] = precondition.solve(G)
        conv , residual = converged(N_i,N_j,u_k,u_k_1,tol)
        residuals.append(residual)
        u_k = u_k_1.copy()

    return u_k_1,iter,residuals

def line_gauss_i(N_i, N_j, beta, array,iter_max,tol):
    conv = False
    u_k = array.copy()
    # BC for next step
    u_k_1 = array.copy()
    B = np.diag((-2 * (1 + beta**2)) * np.ones(N_j - 2)) + np.diag(np.ones(N_j - 3), -1) + np.diag(np.ones(N_j - 3), 1)
    precondition = spilu(csc_matrix(B))
    iter = 0
    residuals = []

    while not conv and iter < iter_max:
        iter += 1
        for i in range(1, N_i - 1):
            G = -beta**2 * (u_k_1[1:-1, i - 1] + u_k[1:-1, i + 1])
            G[0] -= u_k_1[0, i]
            G[-1] -= u_k_1[-1, i]
            u_k_1[1:-1, i] = precondition.solve(G)
        conv, residual = converged(N_i, N_j, u_k, u_k_1,tol)
        residuals.append(residual)
        u_k = u_k_1.copy()

    return u_k_1, iter, residuals

@jit(nopython=True)
def SOR(N_i,N_j,beta,array,omega,iter_max,tol):
    conv = False
    u_k = array.copy()
    # BC for next step
    u_k_1 = array.copy()
    iter = 0
    residuals = []

    while not conv and iter <iter_max:
        iter += 1
        for i in range(1,N_j-1):
            for j in range(1,N_i-1):
                u_k_1[i,j] = (1-omega)*u_k[i,j] + omega/(2*(1+beta**2)) * (
                u_k_1[i-1,j] + u_k[i+1,j] + beta**2 * (u_k_1[i,j-1]+u_k[i,j+1]) 
                )
                
        conv , residual = converged(N_i,N_j,u_k,u_k_1,tol)
        residuals.append(residual)
        u_k = u_k_1.copy()

    return u_k_1 , iter, residuals

def line_SOR(N_i,N_j,beta,array,omega,iter_max,tol):
    conv = False
    u_k = array.copy()
    # BC for next step
    u_k_1 = array.copy()
    B = np.diag((-2*(1+beta**2)) * np.ones(N_i-2)) + np.diag(omega * np.ones(N_i-3), -1) + np.diag(omega * np.ones(N_i-3), 1)
    precondition = spilu(csc_matrix(B))
    iter = 0
    residuals = []

    while not conv and iter <iter_max:
        iter += 1 
        for j in range(1,N_j-1):
            G = -1*(1-omega)*(2*(1+beta**2))*u_k[j,1:-1] - (omega * (beta**2 * (u_k_1[j-1,1:-1] + u_k[j+1,1:-1])))
            G[0] -= omega * u_k_1[j,0]
            G[-1] -= omega * u_k_1[j,-1]
            u_k_1[j,1:-1] = precondition.solve(G)
        conv , residual = converged(N_i,N_j,u_k,u_k_1,tol)
        residuals.append(residual)
        u_k = u_k_1.copy()

    return u_k_1 , iter, residuals

Extension Work: Stochastic Gradient Decent approach¶

# Function being minimised
# In this case we are using the number of iterations to be the value we are trying to mimise
def f(x):
    _,iter,_ = line_SOR(N_i,N_j,beta,initial,x,iter_max,tol)
    return iter

# Approximate derivative of the function since can't calculate analytical value
def approximate_df(x, h=0.0002):
    return (f(x + h) - f(x)) / h

def SGD(start_range, end_range, target_step_size=0.01, iterations=16, gradient_threshold=1e-5):
    np.random.seed(42)  # Ensure reproducibility

    # Random Starting point in range given
    x = np.random.uniform(start_range, end_range)

    for i in range(iterations):
        # Calculating approx gradient
        grad = approximate_df(x)
        
        # Compute the initial update using a learning rate that contributes to a step size of 0.01
        learning_rate_adjusted = target_step_size / (np.abs(grad) + 1e-6)  # Avoid division by zero
        stochastic_factor_adjusted = target_step_size * 0.1  # Adjust stochastic factor relative to target step size

        # Updating omega value
        update = -learning_rate_adjusted * grad + np.random.uniform(-stochastic_factor_adjusted, stochastic_factor_adjusted)
        
        # Scale the update if necessary to ensure its magnitude is roughly target_step_size
        # Scaling update to make sure in correct ballpark of 0.01
        update_magnitude = np.sqrt(update**2)
        scaled_update = (update / update_magnitude) * target_step_size if update_magnitude > 0 else 0

        # Making adjustments and checking if suitable
        x_new = x + scaled_update
        x_new = min(max(x_new, start_range), end_range)

        # Print progress
        print(f"Iteration {i}, omega = {x_new}, Iterations = {f(x_new)}")

        x = x_new  # Update x for the next iteration

    return x
