def upwind(initial,dt,dx):
    c = (300*dt)/dx
    N_dt = round(0.45/dt)
    U_array = np.zeros((len(initial),N_dt+1))
    U_array[:,0] = initial


    for i in range(1,N_dt+1):
        U_array[1:-1,i] = U_array[1:-1,i-1] - c * (U_array[1:-1,i-1] - U_array[0:-2,i-1])

    return U_array

def lax(initial,dt,dx):
    c = (300*dt)/dx
    N_dt = round(0.45/dt)
    U_array = np.zeros((len(initial),N_dt+1))
    U_array[:,0] = initial


    for i in range(1,N_dt+1):
        U_array[1:-1,i] = U_array[1:-1,i-1] - c/2 * (U_array[2:,i-1] - U_array[0:-2,i-1]) + (c**2)/2 * (U_array[0:-2,i-1] - 2*U_array[1:-1,i-1] + U_array[2:,i-1])

    return U_array

def Euler(initial,dt,dx):
    c = (300*dt)/dx
    N_dt = round(0.45/dt)
    N_i = len(initial)
    U_array = np.zeros((len(initial),N_dt+1))
    U_array[:,0] = initial

    # Correctly create a tridiagonal matrix B for the interior points
    B = np.zeros((N_i-2, N_i-2))
    np.fill_diagonal(B, -1)  # Main diagonal
    np.fill_diagonal(B[:-1, 1:], -c / 2)  # Upper diagonal
    np.fill_diagonal(B[1:, :-1], c / 2)  # Lower diagonal
    precondition = spilu(csc_matrix(B))

    for i in range(1,N_dt+1):
        G = -1 * U_array[1:-1,i-1]
        U_array[1:-1,i] = precondition.solve(G)

    return U_array

def lax_2(initial,dt,dx,t_final):
    N_dt = round(t_final/dt)
    U_array = np.zeros((len(initial),N_dt+1))
    U_array[:,0] = initial
    U_array[0,:] = 1

    for i in range(1,N_dt+1):
        U_array[1:-1,i] = 0.5 * (U_array[2:,i-1] + U_array[0:-2,i-1]) - (dt/(2*dx)) * ((U_array[2:,i-1])**2/2 - (U_array[0:-2,i-1])**2/2)

def MacCormack(initial,dt,dx,t_final):
    N_dt = round(t_final/dt)
    U_array = np.zeros((len(initial),N_dt+1))
    U_array[:,0] = initial
    U_array[0,:] = 1
    
    for i in range(1,N_dt+1):
      u1 = np.zeros((len(initial)))
      u1[0] = 1

      # Predictor-step
      u1[1:-1] = U_array[1:-1,i-1] - (dt/dx) * ((U_array[2:,i-1])**2 / 2 - (U_array[1:-1,i-1])**2 / 2)
    
      #Corrector-step      
      U_array[1:-1,i] = 0.5 * ((U_array[1:-1,i-1] + u1[1:-1]) -  (dt/dx) * ((u1[1:-1])**2 / 2 - (u1[0:-2])**2 / 2))
      
    return U_array
